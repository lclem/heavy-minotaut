94  static StateType SanitizeAutsForInclusion(
95  Automaton& smaller,

97  {
98  StateType stateCnt = 0;

100  StateToStateTranslWeak stateTrans(stateMap,

102 
103  Automaton tmpAut = smaller.RemoveUselessStates();
104  Automaton newSmaller = tmpAut.ReindexStates(stateTrans);
105 
106  tmpAut = bigger.RemoveUselessStates();
107  stateMap.clear();
108  Automaton newBigger = tmpAut.ReindexStates(stateTrans);
109 
110  smaller = newSmaller;
111  bigger = newBigger;
112 
113  return stateCnt;
114  }

167  bool operator<(const TTransition& rhs) const
168  {
169  if (parent_ < rhs.parent_)
170  {
171  return true;
172  }
173  else if (parent_ == rhs.parent_)
174  {
175  if (symbol_ < rhs.symbol_)
176  {
177  return true;
178  }
179  else if (symbol_ == rhs.symbol_)
180  {
181  if (children_.size() != rhs.children_.size())
182  {
183  return children_.size() < rhs.children_.size();
184  }
185  else
186  {
187  for (size_t i = 0; i < children_.size(); ++i)
188  {
189  if (children_[i] != rhs.children_[i])
190  {
191  return children_[i] < rhs.children_[i];
192  }
193  }
194  }
195  }
196  }
197 
198  return false;
199  }
200 
201 
202  bool operator==(const TTransition& rhs) const
203  {
204  if ((parent_ == rhs.parent_) && (symbol_ == rhs.symbol_))
205  {
206  if (children_.size() == rhs.children_.size())
207  {
208  size_t i;
209  for (i = 0; i < children_.size(); ++i)
210  {
211  if (children_[i] != rhs.children_[i])
212  {
213  break;
214  }
215  }

218  {
219  return true;
220  }
221  }
