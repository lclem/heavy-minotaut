171  { // check all candidates for 'p'

173  if (data_.end() == iter)
174  { // in the case there is no pair (p, _) in the antichain
175  continue;

208  template <class Cont, class Cmp, class Eraser = DummyEraser>
209  void refine(
210  const Cont& candidates,
211  const T& Q,
212  const Cmp& cmp,
213  const Eraser& eraser = DummyEraser())
214  {
215  for (const Key& p : candidates)
216  { // check all candidates for 'p'
217  auto iter = data_.find(p);
218  if (data_.end() == iter)
219  { // in the case there is no pair (p, _) in the antichain
220  continue;
221  }
222 
223  for (auto jter = iter->second.begin(); jter != iter->second.end(); )
224  { // for all 'P' such that (p, P) is in the antichain, check whether P <= Q
225  auto kter = jter++;
226 
227  if (cmp(*kter, Q))
228  { // if P <= Q, erase (p, P) from the antichain
229  eraser(p, kter);
230 
231  // we rely on the fact that erase() does not invalidate jter
232  iter->second.erase(kter);
233  }
234  }
235 
236  if (iter->second.empty())
237  { // in case there is no (p, _) left, remove 'p'
238  data_.erase(iter);
239  }
240  }
241  }
