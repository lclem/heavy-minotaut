221  inline void SetIthVariableValue(size_t i, char value)
222  {
223  // Assertions
224  assert(i < length());
225  assert((value == ZERO) || (value == ONE) || (value == DONT_CARE));
226 
227  // prepare the mask
228  char mask = (DefaultMask << getIndexInsideChar(i)) ^ static_cast<char>(-1);
229 
230  // mask out bits at given position by the mask
231  vars_[getIndexOfChar(i)] &= mask;
232 
233  // prepare new value of given bits
234  value <<= getIndexInsideChar(i);
235 
236  // insert the value of given bits
237  vars_[getIndexOfChar(i)] |= value;
238  }

333  {
334  if ((lhs.length() < rhs.length()) || rhs.length() < lhs.length())
335  {
336  return lhs.length() < rhs.length();
337  }
338 
339  for (size_t i = 0; i < lhs.length(); ++i)
340  {
341  char lhsIthValue = lhs.GetIthVariableValue(lhs.length() - i - 1);
342  char rhsIthValue = rhs.GetIthVariableValue(rhs.length() - i - 1);
343 
344  switch (lhsIthValue)
345  {
346  case SymbolicVarAsgn::ZERO:
347  switch (rhsIthValue)
348  {
349  case SymbolicVarAsgn::ZERO: continue; break;
350  case SymbolicVarAsgn::ONE: return true; break;

353  }
